[101]
name = 貂禅
desc = 貂婵是汉献帝的大臣王允的歌妓，她不但貌似明月，且能歌善舞。书上说她：“年方二八，色伎俱佳”。绝不亚于昭君、西施，是无可撼动的绝世美人。只是历史上真的是否有貂蝉，史籍没有明确记载。

[102]
name = 甄宓
desc = 历史上记载真有其人，甄宓，袁绍儿媳，袁熙之妻。书上说她：“玉肌花貌，有倾国之色”。如诗如画，如九天神女一样超凡脱俗的仙子。相传曹子建《洛神赋》中的洛水之神，就是他的嫂嫂甄宓。

[103]
name = 小乔
desc = 小乔，乔玄之女，和其姊小乔并称“江东二乔”。姐妹一双生得貌若天仙，堪称绝代佳人。大乔嫁孙策，小乔嫁周瑜。俏丽可人，妩媚动人型的绝世佳人。曹子建赋云：“揽二乔于东南兮，乐朝夕之与共。”

[104]
name = 大乔
desc = 大乔，乔玄之女，孙策之妻，绝色美人。有沉鱼之容，尤其女红一事更是名闻遐迩。和其妹小乔并称“江东二乔”。温柔大体，秀外慧中，是坚贞传统型的古典淑女。

[105]
name = 孙尚香
desc = 孙坚之女，孙策、孙权的同父异母妹妹，后嫁于刘备，也就是“赔了夫人又折兵”里的女主角。性情刚烈，却侠义忠贞，温柔时如细水绵绵，颇有江湖女侠的风范。

[106]
name = 蔡琰
desc = 蔡琰字文姬，是东汉大文学家蔡邕的女儿。初嫁卫氏，夫亡无子，归宁于家。兵乱中被虏，被胡兵辗转掳入南匈奴。性情温柔，才智过人，琴棋书画，无一不精，三国第一才女非蔡琰莫属。

[107]
name = 邹氏
desc = 邹氏，张济之妻，在宛城攻略战中被曹操看上，结果送了典韦一条性命，可见红颜祸水，为害之甚。天香国色，姿色撩人，狐媚妖娆的丰韵佳人。

[108]
name = 樊氏
desc = 樊氏，桂阳守将赵范之嫂，书上说：“子龙见妇人身穿缟素，有倾国倾城之色。”绝色娇艳，芙蓉出水般的清丽惑人，如水一样的娇艳迷人。

[109]
name = 杜氏
desc = 杜氏是吕布麾下秦宜禄之妻，长得非常漂亮，关羽也曾暗恋她。吕布殒命前，关羽曾向曹操讨要杜氏，曹操很爽快就答应了。后发现这个女人美得不寻常，曹操实在舍不得给关羽，遂自纳之。

[110]
name = 吴氏
desc = 吴懿的妹妹，原先吴懿事刘璋的时候，她给刘璋做了儿媳。后来刘备打入成都，便把她拿来做了自己的老婆，刘备称帝后，吴氏被册封为皇后。温柔贤淑，如玉美人，美貌非凡。
